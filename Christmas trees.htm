  <script>

    // Snow generator

    const snow = document.getElementById('snow');

    const N = 55;

    let running = true;

 

    function rand(min, max){ return Math.random()*(max-min)+min; }

 

    for(let i=0;i<N;i++){

      const f = document.createElement('div');

      f.className = 'flake';

      const size = rand(2.5, 7.5);

      const left = rand(0, 100);

      const dur = rand(6, 14);

      const delay = rand(-12, 0);

      const dx = `${rand(-60,60)}vw`;

      f.style.width = f.style.height = size+'px';

      f.style.left = left+'vw';

      f.style.opacity = rand(0.35, 0.95);

      f.style.animationDuration = dur+'s';

      f.style.animationDelay = delay+'s';

      f.style.setProperty('--dx', dx);

      snow.appendChild(f);

    }

 

    // Controls

    const toggleBtn = document.getElementById('toggle');

    toggleBtn.onclick = () => {

      running = !running;

      snow.style.display = running ? 'block' : 'none';

      toggleBtn.textContent = running ? '暂停雪花' : '开启雪花';

    };

 

    const msgs = [

      "愿你平安喜乐 ✨",

      "圣诞快乐，心想事成 🎁",

      "愿温暖与你常在 ☕",

      "新的一年顺顺利利 🌟",

      "好运和你一起“叮叮当” 🔔"

    ];

    const note = document.getElementById('note');

    const changeBtn = document.getElementById('change');

    changeBtn.onclick = () => {

      note.textContent = msgs[Math.floor(Math.random()*msgs.length)];

    };

  </script>
